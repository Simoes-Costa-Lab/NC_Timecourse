colData <- data.frame(row.names=colnames(dat), data=colnames(dat)) colData <- colData %>% mutate(Condition = factor(ifelse(grepl(pattern = |pGFP, x = data), ,))) colData <- colData %>% mutate(Time = factor(x = str_extract(word(colData$data, sep = _,1,1), pattern =[0-9]+), levels = c(6,8,10,12,14,16))) colData <- colData %>% mutate(Group = factor(paste0(Condition,_,Time)))

Note, this is pairwise comparison between NC and WE cells at each timepoint, additional analysis will be done for detecting genes varying through time. ImpulseDE2

rr dds <- DESeqDataSetFromMatrix(countData = dat, colData = colData, design= ~0 + Group)

#Filter out bad samples (from PCA analysis) bad_samples <- c(\6_pos_lib_ATCACG, \8_pos_lib_CGATGT, \12_pos_ASH_CGTACG,\10_pos_lib_TTAGGC, 10_mGFP_1_ATGTCA, \6_neg_lib_TGACCA, \8_neg_lib_ACAGTG, 12_mGFP_1_GTGAAA) dds <- dds[,!(colnames(dds) %in% bad_samples)]

#Filter out genes lowly detected. keep <- rowSums(counts(dds)) >= 25 dds <- dds[keep,]

dds <- DESeq(dds, parallel = TRUE)
